30 Key DualUse::specialAction(uint8_t spec_index) {
31  Key new_key;
32 
33  new_key.flags = KEY_FLAGS;
34  if (spec_index < 8) {
35  new_key.keyCode = Key_LeftControl.keyCode + spec_index;
36  } else {
37  uint8_t target = spec_index - 8;
38 
39  Layer.on(target);
40 
41  new_key.keyCode = 0;
42  }
43 
44  return new_key;
45 }

47 void DualUse::pressAllSpecials(byte row, byte col) {
48  for (uint8_t spec_index = 0; spec_index < 32; spec_index++) {
49  if (!bitRead(pressed_map_, spec_index))
50  continue;
51 
52  Key new_key = specialAction(spec_index);
53  if (new_key.raw != Key_NoKey.raw)
54  handleKeyswitchEvent(new_key, row, col, IS_PRESSED | INJECTED);
55  }
56 }

73 Key DualUse::eventHandlerHook(Key mapped_key, byte row, byte col, uint8_t key_state) {
74  if (key_state & INJECTED)
75  return mapped_key;
76 
77  // If nothing happened, bail out fast.
78  if (!keyIsPressed(key_state) && !keyWasPressed(key_state)) {
79  if (mapped_key.raw < ranges::DU_FIRST || mapped_key.raw > ranges::DU_LAST)
80  return mapped_key;
81  return Key_NoKey;
82  }
83 
84  if (mapped_key.raw >= ranges::DU_FIRST && mapped_key.raw <= ranges::DU_LAST) {
85  uint8_t spec_index = (mapped_key.raw - ranges::DU_FIRST) >> 8;
86  Key new_key = Key_NoKey;
87 
88  if (keyToggledOn(key_state)) {
89  bitWrite(pressed_map_, spec_index, 1);
90  bitWrite(key_action_needed_map_, spec_index, 1);
91  end_time_ = millis() + time_out;
92  } else if (keyIsPressed(key_state)) {
93  if (millis() >= end_time_) {
94  new_key = specialAction(spec_index);
95  }
96  } else if (keyToggledOff(key_state)) {
97  if ((millis() >= end_time_) && bitRead(key_action_needed_map_, spec_index)) {
98  uint8_t m = mapped_key.raw - ranges::DU_FIRST - (spec_index << 8);
99  if (spec_index >= 8)
100  m--;
101 
102  Key new_key = { m, KEY_FLAGS };
103 
104  handleKeyswitchEvent(new_key, row, col, IS_PRESSED | INJECTED);
105  hid::sendKeyboardReport();
106  } else {
107  if (spec_index >= 8) {
108  uint8_t target = spec_index - 8;
109 
110  Layer.off(target);
111  }
112  }
113 
114  bitWrite(pressed_map_, spec_index, 0);
115  bitWrite(key_action_needed_map_, spec_index, 0);
116  }
117 
118  return new_key;
119  }
120 
121  if (pressed_map_ == 0) {
122  return mapped_key;
123  }
124 
125  pressAllSpecials(row, col);
126  key_action_needed_map_ = 0;
127 
128  if (pressed_map_ > (1 << 7)) {
129  mapped_key = Layer.lookup(row, col);
130  }
131 
132  return mapped_key;
133 }
