46 void EEPROMSettings::seal(void) {
47  sealed_ = true;
48 
49  CRC.finalize();
50 
51  if (settings_.magic[0] != 'K' || settings_.magic[1] != 'S') {
52  settings_.magic[0] = 'K';
53  settings_.magic[1] = 'S';
54  settings_.version = 0;
55  settings_.crc = CRC.crc;
56 
57  return update();
58  }
59 
60  if (settings_.crc != CRC.crc)
61  is_valid_ = false;
62 }
