56 void Focus::loopHook(bool is_post_clear) {
57  if (is_post_clear)
58  return;
59 
60  if (Serial.available() == 0)
61  return;
62 
63  uint8_t i = 0;
64  do {
65  command_[i++] = Serial.read();
66 
67  if (Serial.peek() == '\n')
68  break;
69  } while (command_[i - 1] != ' ' && i < 32);
70  if (command_[i - 1] == ' ')
71  command_[i - 1] = '\0';
72  else
73  command_[i] = '\0';
74 
75  for (HookNode *node = root_node_; node; node = node->next) {
76  if ((*node->handler)(command_)) {
77  break;
78  }
79  }
80 
81  Serial.println(F("."));
82 
83  drain();
84 
85  if (Serial.peek() == '\n')
86  Serial.read();
87 }

113 bool Focus::helpHook(const char *command) {
