26 void hsvToRgb(cRGB *cRGB, uint16_t h, uint16_t s, uint16_t v) {
27  /* HSV to RGB conversion function with only integer
28  math */
29  uint16_t region, fpart, p, q, t;
30 
31  if (s == 0) {
32  /* color is grayscale */
33  cRGB->r = cRGB->g = cRGB->b = v;
34  return;
35  }
36 
37  /* make hue 0-5 */
38  region = h / 43;
39  /* find remainder part, make it from 0-255 */
40  fpart = (h - (region * 43)) * 6;
41 
42  /* calculate temp vars, doing integer multiplication */
43  p = (v * (255 - s)) >> 8;
44  q = (v * (255 - ((s * fpart) >> 8))) >> 8;
45  t = (v * (255 - ((s * (255 - fpart)) >> 8))) >> 8;
46 
47  /* assign temp vars based on color cone region */
48  switch (region) {
49  case 0:
50  cRGB->r = v;
51  cRGB->g = t;
52  cRGB->b = p;
53  break;
54  case 1:
55  cRGB->r = q;
56  cRGB->g = v;
57  cRGB->b = p;
58  break;
59  case 2:
60  cRGB->r = p;
61  cRGB->g = v;
62  cRGB->b = t;
63  break;
64  case 3:
65  cRGB->r = p;
66  cRGB->g = q;
67  cRGB->b = v;
68  break;
69  case 4:
70  cRGB->r = t;
71  cRGB->g = p;
72  cRGB->b = v;
73  break;
74  default:
75  cRGB->r = v;
76  cRGB->g = p;
77  cRGB->b = q;
78  break;
79  }
