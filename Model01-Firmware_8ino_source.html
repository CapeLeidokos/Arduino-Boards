184 static void anyKeyMacro(uint8_t keyState) {
185  static Key lastKey;
186  if (keyToggledOn(keyState))
187  lastKey.keyCode = Key_A.keyCode + (uint8_t)(millis() % 36);
188 
189  if (keyIsPressed(keyState))
190  kaleidoscope::hid::pressKey(lastKey);

242 void setup() {
243  // First, call Kaleidoscope's internal setup function
244  Kaleidoscope.setup();
245 
246  // Next, tell Kaleidoscope which plugins you want to use.
247  // The order can be important. For example, LED effects are
248  // added in the order they're listed here.
249  Kaleidoscope.use(
250  // The boot greeting effect pulses the LED button for 10 seconds after the keyboard is first connected
251  &BootGreetingEffect,
252 
253  // The hardware test mode, which can be invoked by tapping Prog, LED and the left Fn button at the same time.
254  &TestMode,
255 
256  // LEDControl provides support for other LED modes
257  &LEDControl,
258 
259  // We start with the LED effect that turns off all the LEDs.
260  &LEDOff,
261 
262  // The rainbow effect changes the color of all of the keyboard's keys at the same time
263  // running through all the colors of the rainbow.
264  &LEDRainbowEffect,
265 
266  // The rainbow wave effect lights up your keyboard with all the colors of a rainbow
267  // and slowly moves the rainbow across your keyboard
268  &LEDRainbowWaveEffect,
269 
270  // The chase effect follows the adventure of a blue pixel which chases a red pixel across
271  // your keyboard. Spoiler: the blue pixel never catches the red pixel
272  &LEDChaseEffect,
273 
274  // These static effects turn your keyboard's LEDs a variety of colors
275  &solidRed, &solidOrange, &solidYellow, &solidGreen, &solidBlue, &solidIndigo, &solidViolet,
276 
277  // The breathe effect slowly pulses all of the LEDs on your keyboard
278  &LEDBreatheEffect,
279 
280  // The AlphaSquare effect prints each character you type, using your
281  // keyboard's LEDs as a display
282  &AlphaSquareEffect,
283 
284  // The stalker effect lights up the keys you've pressed recently
285  &StalkerEffect,
286 
287  // The numlock plugin is responsible for lighting up the 'numpad' mode
288  // with a custom LED effect
289  &NumLock,
290 
291  // The macros plugin adds support for macros
292  &Macros,
293 
294  // The MouseKeys plugin lets you add keys to your keymap which move the mouse.
295  &MouseKeys
296  );
297 
298  // While we hope to improve this in the future, the NumLock plugin
299  // needs to be explicitly told which keymap layer is your numpad layer
300  NumLock.numPadLayer = NUMPAD;
301 
302  // We configure the AlphaSquare effect to use RED letters
303  AlphaSquare.color = { 255, 0, 0 };
304 
305  // We set the brightness of the rainbow effects to 150 (on a scale of 0-255)
306  // This draws more than 500mA, but looks much nicer than a dimmer effect
307  LEDRainbowEffect.brightness(150);
308  LEDRainbowWaveEffect.brightness(150);
309 
310  // The LED Stalker mode has a few effects. The one we like is
311  // called 'BlazingTrail'. For details on other options,
312  // see https://github.com/keyboardio/Kaleidoscope-LED-Stalker
313  StalkerEffect.variant = STALKER(BlazingTrail);
314 
315  // We want to make sure that the firmware starts with LED effects off
316  // This avoids over-taxing devices that don't have a lot of power to share
317  // with USB devices
318  LEDOff.activate();
319 }
