50 Key FingerPainter::eventHandlerHook(Key mapped_key, byte row, byte col, uint8_t key_state) {
51  if (!edit_mode_)
52  return mapped_key;
53 
54  if (!keyToggledOn(key_state))
55  return Key_NoKey;
56 
57  if (row >= ROWS || col >= COLS)
58  return Key_NoKey;
59 
60  uint8_t color_index = ::LEDPaletteTheme.lookupColorIndexAtPosition(color_base_, KeyboardHardware.getLedIndex(row, col));
61 
62  // Find the next color in the palette that is different.
63  // But do not loop forever!
64  bool turn_around = false;
65  cRGB old_color = ::LEDPaletteTheme.lookupPaletteColor(color_index), new_color = old_color;
66  while (memcmp(&old_color, &new_color, sizeof(cRGB)) == 0) {
67  color_index++;
68  if (color_index > 15) {
69  color_index = 0;
70  if (turn_around)
71  break;
72  turn_around = true;
73  }
74  new_color = ::LEDPaletteTheme.lookupPaletteColor(color_index);
75  }
76 
77  ::LEDPaletteTheme.updateColorIndexAtPosition(color_base_, KeyboardHardware.getLedIndex(row, col), color_index);
78 
79  return Key_NoKey;
80 }

82 bool FingerPainter::focusHook(const char *command) {
83  enum {
84  TOGGLE,
85  CLEAR,
86  } sub_command;
87 
88  if (strncmp_P(command, PSTR("fingerpainter."), 14) != 0)
89  return false;
90 
91  if (strcmp_P(command + 14, PSTR("toggle")) == 0)
92  sub_command = TOGGLE;
93  else if (strcmp_P(command + 14, PSTR("clear")) == 0)
94  sub_command = CLEAR;
95  else
96  return false;
97 
98  if (sub_command == CLEAR) {
99  for (uint16_t i = 0; i < ROWS * COLS / 2; i++) {
100  EEPROM.update(color_base_ + i, 0);
101  }
102  return true;
103  }
104 
105  ::FingerPainter.activate();
106  toggle();
107 
108  return true;
109 }
