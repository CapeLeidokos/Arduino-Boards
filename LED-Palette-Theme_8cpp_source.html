82 void LEDPaletteTheme::updateColorIndexAtPosition(uint16_t map_base, uint16_t position, uint8_t color_index) {
83  uint8_t indexes;
84 
85  indexes = EEPROM.read(map_base + position / 2);
86  if (position % 2) {
87  uint8_t other = indexes >> 4;
88  indexes = (other << 4) + color_index;
89  } else {
90  uint8_t other = indexes & ~0xf0;
91  indexes = (color_index << 4) + other;
92  }
93  EEPROM.update(map_base + position / 2, indexes);
94 }

127 bool LEDPaletteTheme::themeFocusHandler(const char *command, const char *expected_command,
128  uint16_t theme_base, uint8_t max_themes) {
129  if (strcmp_P(command, expected_command) != 0)
130  return false;
131 
132  uint16_t max_index = (max_themes * ROWS * COLS) / 2;
133 
134  if (Serial.peek() == '\n') {
135  for (uint16_t pos = 0; pos < max_index; pos++) {
136  uint8_t indexes = EEPROM.read(theme_base + pos);
137 
138  ::Focus.printNumber(indexes >> 4);
139  ::Focus.printSpace();
140  ::Focus.printNumber(indexes & ~0xf0);
141  ::Focus.printSpace();
142  }
143  Serial.println();
144  return true;
145  }
146 
147  uint16_t pos = 0;
148 
149  while ((Serial.peek() != '\n') && (pos < max_index)) {
150  uint8_t idx1 = Serial.parseInt();
151  uint8_t idx2 = Serial.parseInt();
152  uint8_t indexes = (idx1 << 4) + idx2;
153 
154  EEPROM.update(theme_base + pos, indexes);
155  pos++;
156  }
157 
158  return true;
159 }
