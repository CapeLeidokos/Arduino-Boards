46 
47  tapDanceAction(idx, last_tap_dance_row_, last_tap_dance_col_, tap_count_[idx], Interrupt);

107 void TapDance::actionKeys(uint8_t tap_count, ActionType tap_dance_action, uint8_t max_keys, const Key tap_keys[]) {
108  if (tap_count > max_keys)
109  tap_count = max_keys;
110 
111  Key key;
112  key.raw = pgm_read_word(&(tap_keys[tap_count - 1].raw));
113 
114  switch (tap_dance_action) {
115  case Tap:
116  break;
117  case Interrupt:
118  case Timeout:
119  handleKeyswitchEvent(key, last_tap_dance_row_, last_tap_dance_col_, IS_PRESSED | INJECTED);
120  break;
121  case Hold:
122  handleKeyswitchEvent(key, last_tap_dance_row_, last_tap_dance_col_, IS_PRESSED | WAS_PRESSED | INJECTED);
123  break;
124  case Release:
125  hid::sendKeyboardReport();
126  handleKeyswitchEvent(key, last_tap_dance_row_, last_tap_dance_col_, WAS_PRESSED | INJECTED);
127  break;
128  }
129 }

133 Key TapDance::eventHandlerHook(Key mapped_key, byte row, byte col, uint8_t key_state) {
134  if (key_state & INJECTED)
135  return mapped_key;
136 
137  if (!keyIsPressed(key_state) && !keyWasPressed(key_state)) {
138  if (isTapDance(mapped_key))
139  return Key_NoKey;
140 
141  return mapped_key;
142  }
143 
144  if (!isTapDance(mapped_key)) {
145  if (!isActive())
146  return mapped_key;
147 
148  if (keyToggledOn(key_state))
149  interrupt(row, col);
150 
151  if (KeyboardHardware.isKeyMasked(row, col)) {
152  KeyboardHardware.unMaskKey(row, col);
153  return Key_NoKey;
154  }
155  return mapped_key;
156  }
157 
158  uint8_t tap_dance_index = mapped_key.raw - ranges::TD_FIRST;
159 
160  if (keyToggledOff(key_state))
161  bitClear(pressed_state_, tap_dance_index);
162 
163  if (!isInSeq(mapped_key)) {
164  if (!isActive()) {
165  if (bitRead(triggered_state_, tap_dance_index)) {
166  if (keyToggledOff(key_state))
167  return release(tap_dance_index);
168  return Key_NoKey;
169  }
170 
171  last_tap_dance_key_.raw = mapped_key.raw;
172  last_tap_dance_row_ = row;
173  last_tap_dance_col_ = col;
174  return tap();
175  } else {
176  if (keyToggledOff(key_state) && stillHeld(tap_dance_index)) {
177  return release(tap_dance_index);
178  }
179 
180  if (!keyToggledOn(key_state))
181  return Key_NoKey;
182 
183  interrupt(row, col);
184  }
185  }
186 
187  // in sequence
188 
189  if (keyToggledOff(key_state))
190  return Key_NoKey;
191 
192  last_tap_dance_key_.raw = mapped_key.raw;
193  last_tap_dance_row_ = row;
194  last_tap_dance_col_ = col;
195  bitSet(pressed_state_, tap_dance_index);
196 
197  if (keyToggledOn(key_state))
198  return tap();
199 
200  if (bitRead(triggered_state_, tap_dance_index))
201  tapDanceAction(tap_dance_index, row, col, tap_count_[tap_dance_index], Hold);
202 
203  return Key_NoKey;
204 }

206 void
207 TapDance::loopHook(bool is_post_clear) {
208  if (!is_post_clear)
209  return;
210 
211  for (uint8_t i = 0; i < 16; i++) {
212  if (!bitRead(release_next_state_, i))
213  continue;
214 
215  tapDanceAction(i, last_tap_dance_row_, last_tap_dance_col_, tap_count_[i], Release);
216  tap_count_[i] = 0;
217  bitClear(release_next_state_, i);
218  }
219 
220  if (!isActive())
221  return;
222 
223  if (end_time_ && millis() > end_time_)
224  timeout();
225 }
